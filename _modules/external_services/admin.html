class MenuItemAdmin(admin.ModelAdmin):
    search_fields = (
        'course_instance__instance_name',
        'course_instance__course__code',
        'course_instance__course__name',
        'menu_url',
        'menu_label',
        'service__menu_label',
        'service__url',
    )
    list_display_links = (
        'id',
        'menu_label',
    )
    list_display = (
        'id',
        'course_instance',
        'service',
        'menu_url',
        'menu_label',
        'access',
        'enabled',
    )
    list_filter = (
        'enabled',
        'service',
        RecentCourseInstanceListFilter,
    )
    raw_id_fields = (
        'course_instance',
        'service',
    )
