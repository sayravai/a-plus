def set_organization(apps, schema_editor):
    Model = apps.get_model('userprofile', 'UserProfile')
    for profile in Model.objects.all():
        if bool(profile.student_id):
            profile.organization = settings.LOCAL_ORGANIZATION
            profile.save()
